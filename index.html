تنزيل النموذج المكمّم

wget https://huggingface.co/TheBloke/TinyLlama-1.1B-Chat-v1.0-GGUF/resolve/main/tinyllama-1.1b-chat-v1.0.Q4_K_M.gguf
استعراض النماذج المتاحة

تشغيل النموذج واستخدامه

قم بتشغيل النموذج باستخدام الأوامر التالية:

from llama_cpp import Llama

model = Llama("tinyllama-1.1b-chat-v1.0.Q4_K_M.gguf")
response = model.create_chat_completion([
    {"role": "user", "content": "مرحباً، كيف حالك؟"}
])
print(response["choices"][0]["message"]["content"])

واجهة API متكاملة

استخدم واجهة برمجة التطبيقات (API) للوصول إلى نماذج الذكاء الاصطناعي المتقدمة مثل DeepSeek-R1-70B من خلال واجهة RESTful بسيطة:

curl -X POST 'https://api.search1api.com/v1/chat/completions' \
--header 'Content-Type: application/json' \
--data '{
  "model": "deepseek-r1-70b-online",
  "messages": [
    {
      "role": "user",
      "content": "Explain quantum computing in simple terms"
    }
  ],
  "stream": true
}'
import requests
import json

url = "https://api.search1api.com/v1/chat/completions"
headers = {
    "Content-Type": "application/json"
}
data = {
    "model": "deepseek-r1-70b-online",
    "messages": [
        {
            "role": "user",
            "content": "Explain quantum computing in simple terms"
        }
    ],
    "stream": True
}

response = requests.post(url, headers=headers, json=data)
print(response.text)
const url = 'https://api.search1api.com/v1/chat/completions';
const headers = {
  'Content-Type': 'application/json'
};
const data = {
  model: 'deepseek-r1-70b-online',
  messages: [
    {
      role: 'user',
      content: 'Explain quantum computing in simple terms'
    }
  ],
  stream: true
};

fetch(url, {
  method: 'POST',
  headers: headers,
  body: JSON.stringify(data)
})
.then(response => response.text())
.then(text => console.log(text));
